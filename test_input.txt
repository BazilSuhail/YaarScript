enum Color { RED, GREEN, BLUE };

const int TARGET = 5;

// Function to print a separator line
void print_separator(int count) {

    print("\n");
    for (int i = 0; i < TARGET; i++) {
        print("=");
    }
    print("\n");
}

main {
    // Example of the new features:
    print("Enter Number: ");

    int base = read();
    int power = 3;
    print("Power for ", base, " is ", base ** power);

    int start = time();
    print("\n", random(1, 100));
    int end = time();
    print("\nTime taken: ", end - start);

    print("--- 🚀 THE ULTIMATE LANGUAGE POWER TEST ---\n");

    // 1. The Numerical Diamond (Nested For Loops)
    //int size = 5 + 'x';
    int size = 5 ;
    //int size1 = 5 + "sd";
    print("\n[1] Generating Numeric Diamond:\n");
    
    // Top Half
    for (int i = 1; i <= size; i++) {
        for (int s = 1; s <= (size - i); s++) { print(" "); }
        for (int j = 1; j <= i; j++) { 
            print(" ");
            print(i); 
         }
        print("\n");
    }
    // Bottom Half
    for (int i = (size - 1); i >= 1; i--) {
        for (int s = 1; s <= (size - i); s++) { print(" "); }
        for (int j = 1; j <= i; j++) {
            print(" ");
            print(i); 
         }
        print("\n");
    }

    print_separator(40);

    // 2. Do-While Loop (Diagnostics)
    print("[2] Running Sensor Calibration:\n");
    int attempt = 1;
    bool sensor_ready = false;

    do {
        print("Probe #", attempt, ":\t");
        if (attempt < 10) {
            print("Warming up...\t[WAIT...]\n");
        } else {
            print("Calibration Complete!\t[READY]\n");
            sensor_ready = true;
        }
        attempt = attempt + 1;
    } while (!sensor_ready);

    print_separator(40);

    // 3. Enum & Switch Implementation
    print("[3] System Final Status:\n");
    int status_color = GREEN;
    
    switch (status_color) {
        case RED:   print("SYSTEM FATALcls (RED)\n"); break;
        case GREEN: print("SYSTEM OPTIMAL (GREEN)\n"); break;
        case BLUE:  print("SYSTEM RECOVERY (BLUE)\n"); break;
    }
}
